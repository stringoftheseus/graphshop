function importArcs(arcs)
{
	for(var a=0; a<arcs.length; a++)
	{
		graph.addArc(arcs[a][0], arcs[a][1]);
	}

}

var arcsA = [[0,1], [2,0], [0,3], [4,0], [0,5], [6,0], [1,2], [3,1], [1,4], [5,1], [1,6], [2,3], [4,2], [2,5], [6,2], [3,4], [5,3], [3,6], [4,5], [6,4], [5,6]];

importArcs(arcsA);



var importArcLine = function(c, arcs)
{
	for(var u=0, a=0; u<c; u++)
	{
		for(var v=u+1; v<c; v++, a++)
		{
			if(arcs[a])
			{
				graph.addArc(u, v);
			}
			else
			{
				graph.addArc(v, u);
			}
		}
	}	
}

importArcLine(4, "000101");
